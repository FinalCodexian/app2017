
IF OBJECT_ID ('dbo.MENU_USUARIO') IS NOT NULL
	DROP TABLE dbo.MENU_USUARIO
GO

CREATE TABLE dbo.MENU_USUARIO
	(
	MENU         INT NOT NULL,
	USUARIO      CHAR (5) NOT NULL,
	BASE         VARCHAR (20) NOT NULL,
	LECTURA      CHAR (1) DEFAULT ('N') NULL,
	ESCRITURA    CHAR (1) DEFAULT ('N') NULL,
	CREADOR      VARCHAR (5) NOT NULL,
	FEC_CREACION DATETIME DEFAULT (getdate()) NULL,
	MODIFICA     VARCHAR (5) NULL,
	FEC_MODIFICA DATETIME DEFAULT (NULL) NULL,
	CONSTRAINT PK__MENU_USU__5413CDC791EF3A1E PRIMARY KEY (MENU,USUARIO)
	)
GO
